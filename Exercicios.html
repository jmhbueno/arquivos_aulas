Importar arquivo

link_imdb <- "https://github.com/jmhbueno/Arquivos_aulas/raw/main/imdb.xlsx"

## Importa o arquivo do github para o computador (fora do R)
download.file(
  url = link_imdb,
  destfile = "imdb.xlsx",
  mode = "wb")

# para evitar um problema com a coluna notas_imdb 
imdb <- read_xlsx("imdb.xlsx",)  

# selecionar apenas o primeiro gênero
imdb <- 
  read_xlsx("imdb.xlsx") %>% 
  as.data.frame() %>%
  mutate(generos = str_extract(generos, "^[^|]+"))
